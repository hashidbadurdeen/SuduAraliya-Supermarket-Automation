// C++ program to illustrate Library Function
#include <iostream>
#include <cmath>
using namespace std;
int main() {
	double number, squareRoot;
	cout << "Enter a Number:";
	cin >> number;

	squareRoot = sqrt(number);
	cout << "Square root of " << number << " = " << squareRoot << endl;

}
